    </article>
  );
}

export function SummaryCard({ label, value, caption, badge, badgeVariant = 'positive' }) {
  return (
    <article className="summary-card ds-stack">
      <header className="summary-card-header">
        <p className="summary-card-label">{label}</p>
        {badge ? (
          <span className={`ds-badge summary-card-badge--${badgeVariant}`}>{badge}</span>
        ) : null}
      </header>
      <p className="summary-card-value">{value}</p>
      <p className="summary-card-caption">{caption}</p>
    </article>
  );
}
<template>
  <article class="summary-card ds-stack">
    <header class="summary-card-header">
      <p class="summary-card-label">{{ label }}</p>
      <span v-if="badge" class="ds-badge" :class="`summary-card-badge--${badgeVariant}`">{{ badge }}</span>
    </header>
    <p class="summary-card-value">{{ value }}</p>
    <p class="summary-card-caption">{{ caption }}</p>
  </article>
</template>

<script setup>
defineProps({
  label: String,
  value: [String, Number],
  caption: String,
  badge: String,
  badgeVariant: {
    type: String,
    default: 'positive'
  }
});
</script>

export function DataTable({ rows }) {
  return (
    <div className="table-shell overflow-hidden rounded-3xl border border-slate-100">
      <table className="min-w-full divide-y divide-slate-100 text-sm">
        <thead className="bg-slate-50 text-xs font-semibold uppercase tracking-wider text-slate-500">
          <tr>
            <th scope="col" className="px-6 py-4 text-left">Contato</th>
            <th scope="col" className="px-6 py-4 text-left">Etapa</th>
            <th scope="col" className="px-6 py-4 text-left">Última atividade</th>
          </tr>
        </thead>
        <tbody className="divide-y divide-slate-100 bg-white">
          {rows.map((row) => (
            <tr key={row.id} className="transition hover:bg-slate-50">
              <td className="px-6 py-4">{row.contact}</td>
              <td className="px-6 py-4">{row.stage}</td>
              <td className="px-6 py-4 text-slate-500">{row.lastActivity}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}
<template>
